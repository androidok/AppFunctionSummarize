		 */
		String date = "2001-03-15 15-37-05";
		SimpleDateFormat simpleDateFormat = new SimpleDateFormat("yyyy-MM-dd HH-mm-ss");//24小时制

		long time2 = simpleDateFormat.parse(date).getTime();
